---
title: Agent Observability Provider
description: Context provider that every other component from this library expects. Install and mount it once, high in your tree — it creates the API client, resolves the current session, and shares the hooks that drive the rest of the UI. No visual output of its own.
---

## Installation

<InstallTabs pkg="observability-provider" />

## Usage

```tsx
import { AgentObservabilityProvider } from '@/lib/observability-provider'
import { SessionDetailPage } from '@/components/session-detail-page'

// Wrap the subtree that needs session data. Pass a sessionId when
// rendering a single-session page; list pages can drop it.
export function App({ sessionId }: { sessionId: string }) {
  return (
    <AgentObservabilityProvider
      baseUrl="https://observability.example.com/api"
      sessionId={sessionId}
    >
      <SessionDetailPage />
    </AgentObservabilityProvider>
  )
}
```
