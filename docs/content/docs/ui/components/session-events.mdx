## Usage

```tsx
import { SessionEvents } from '@/components/session-events'

// Default badge palette
<SessionEvents />

// Recolor a specific event type
<SessionEvents
  typeBadgeClass={{
    function_call: 'bg-orange-100 text-orange-800',
  }}
/>
```
