---
title: useSessions
description: Paginated session list fetcher. Holds its own cursor — pass a stable initialOffset and drive pagination with setOffset. Refetches automatically whenever filters or offset change.
---

## Signature

```ts
useSessions(limit?: number, initialOffset?: number, filters?: SessionsFilters): { sessions, meta, loading, error, offset, setOffset }
```

## Installation

<InstallTabs pkg="observability-hooks" />

## Usage

```tsx
import { useSessions } from '@/lib/observability-hooks'

export function MyList() {
  const { sessions, meta, loading, offset, setOffset } = useSessions(20, 0, {
    accountId: 'acc_demo',
  })
  if (loading) return <Spinner />
  return (
    <>
      {sessions.map((s) => (
        <Row key={s.id} session={s} />
      ))}
      <Pagination
        offset={offset}
        limit={meta.limit}
        total={meta.total_count}
        onChange={setOffset}
      />
    </>
  )
}
```
