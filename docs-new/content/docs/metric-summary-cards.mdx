## Usage

```tsx
import { MetricSummaryCards } from '@/components/metric-summary-cards'

// Inside a provider tree — reads the current session via context
<MetricSummaryCards />

// Or pass your own metrics
<MetricSummaryCards metrics={customMetrics} />
```
